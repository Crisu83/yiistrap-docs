12 /**
13  * Bootstrap alert widget.
14  * @see http://twitter.github.com/bootstrap/javascript.html#alerts
15  */
16 class TbAlert extends TbWidget
17 {
18     /**
19      * @var array the alerts configurations (style=>config).
20      */
21     public $alerts;
22     /**
23      * @var string|boolean the close link text. If this is set false, no close link will be displayed.
24      */
25     public $closeText = TbHtml::CLOSE_TEXT;
26     /**
27      * @var boolean indicates whether the alert should be an alert block. Defaults to 'true'.
28      */
29     public $block = true;
30     /**
31      * @var boolean indicates whether alerts should use transitions. Defaults to 'true'.
32      */
33     public $fade = true;
34     /**
35      * @var string[] the JavaScript event configuration (name=>handler).
36      */
37     public $events = array();
38     /**
39      * @var array the HTML attributes for the alert container.
40      */
41     public $htmlOptions = array();
42 
43     /**
44      * Initializes the widget.
45      */
46     public function init()
47     {
48         $this->htmlOptions = TbHtml::defaultOption('id', $this->getId(), $this->htmlOptions);
49         if (is_string($this->alerts))
50             $styles = explode(' ', $this->alerts);
51         else if (!isset($this->alerts))
52             $styles = TbHtml::$alertStyles; // render all styles by default
53         if (isset($styles))
54         {
55             $this->alerts = array();
56             foreach ($styles as $style)
57                 $this->alerts[$style] = array();
58         }
59     }
60 
61     /**
62      * Runs the widget.
63      */
64     public function run()
65     {
66         /* @var $user CWebUser */
67         $user = Yii::app()->getUser();
68         echo CHtml::openTag('div', $this->htmlOptions);
69         foreach ($this->alerts as $style => $alert)
70         {
71             if (isset($alert['visible']) && !$alert['visible'])
72                 continue;
73 
74             if ($user->hasFlash($style))
75             {
76                 $htmlOptions = TbHtml::popOption('htmlOptions', $alert, array());
77                 $htmlOptions = TbHtml::defaultOption('closeText', $this->closeText, $htmlOptions);
78                 $htmlOptions = TbHtml::defaultOption('block', $this->block, $htmlOptions);
79                 $htmlOptions = TbHtml::defaultOption('fade', $this->fade, $htmlOptions);
80                 echo TbHtml::alert($style, $user->getFlash($style), $htmlOptions);
81             }
82         }
83         echo '</div>';
84         $this->registerEvents("#{$this->htmlOptions['id']} > .alert", $this->events);
85     }
86 }
