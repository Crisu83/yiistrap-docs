148     /**
149      * Registers a specific Bootstrap plugin using the given selector and options.
150      * @param string $name the plugin name.
151      * @param string $selector the CSS selector.
152      * @param array $options the JavaScript options for the plugin.
153      * @param int $position the position of the JavaScript code.
154      */
155     public function registerPlugin($name, $selector, $options = array(), $position = CClientScript::POS_END)
156     {
157         $options = !empty($options) ? CJavaScript::encode($options) : '';
158         $script = "jQuery('{$selector}').{$name}({$options});";
159         Yii::app()->clientScript->registerScript($this->getUniqueScriptId(), $script, $position);
160     }

162     /**
163      * Registers events using the given selector.
164      * @param string $selector the CSS selector.
165      * @param string[] $events the JavaScript event configuration (name=>handler).
166      * @param int $position the position of the JavaScript code.
167      */
168     public function registerEvents($selector, $events, $position = CClientScript::POS_END)
169     {
170         if(empty($events))
171             return;
172 
173         $script = '';
174         foreach ($events as $name => $handler)
175         {
176             $handler = ($handler instanceof CJavaScriptExpression)
177                 ? $handler
178                 : new CJavaScriptExpression($handler);
179 
180             $script .= "jQuery('{$selector}').on('{$name}', {$handler});";
181         }
182         Yii::app()->clientScript->registerScript($this->getUniqueScriptId(), $script, $position);
183     }

185     /**
186      * Returns the url to the published assets folder.
187      * @return string the url.
188      */
189     protected function getAssetsUrl()
190     {
191         if (isset($this->_assetsUrl))
192             return $this->_assetsUrl;
193         else
194         {
195             $assetsPath = Yii::getPathOfAlias('bootstrap.assets');
196             $assetsUrl = Yii::app()->assetManager->publish($assetsPath, true, -1, $this->forceCopyAssets);
197             return $this->_assetsUrl = $assetsUrl;
