162     /**
163      * Normalizes the tab configuration.
164      * @param array $tabs the tab configuration
165      * @param array $panes a reference to the panes array
166      * @param integer $i the current index
167      * @return array the items
168      */
169     protected function normalizeTabs($tabs, &$panes, &$i = 0)
170     {
171 
172         $id = TbHtml::getOption('id', $this->htmlOptions, $this->getId());
173         $items = array();
174 
175         //Check if has an active item
176         $hasActiveItem = false;
177         foreach ($tabs as $tab)
178         {
179             if ($hasActiveItem = TbHtml::getOption('active', $tab, false) === true)
180                 break;
181         }
182 
183         foreach ($tabs as $tab)
184         {
185             $item = $tab;
186 
187             if (isset($item['visible']) && $item['visible'] === false)
188                 continue;
189 
190             // if no tab should be active, activate first
191             if (!$hasActiveItem && $i == 0)
192                 $item['active'] = true;
193 
194             // set the label to the title if any
195             if (isset($item['title']))
196                 $item['label'] = TbHtml::defaultOption('label', TbHtml::popOption('title', $item), $item);
197 
198             $item = TbHtml::defaultOption('itemOptions', array(), $item);
199 
200             if (isset($tab['items']))
201             {
202                 $item['linkOptions']['data-toggle'] = 'dropdown';
203                 $item['items'] = $this->normalizeTabs($item['items'], $panes, $i);
204             }
205             else
206             {
207                 $item['linkOptions']['data-toggle'] = 'tab';
208 
209                 $item = TbHtml::defaultOption('id', $id . '_tab_' . ($i + 1), $item);
210 
211                 $item['url'] = '#' . $item['id'];
212 
213                 // no content and we have a view?
214                 if (!isset($item['content']) && isset($item['view']))
215                 {
216                     $view = TbHtml::popOption('view', $item, '');
217 
218                     $data = TbHtml::popOption('data', $item, array());
219                     if (is_array($this->viewData))
220                         $data = TbHtml::mergeOptions($this->viewData, $data);
221 
222                     $process = TbHtml::popOption('processOutput', $item, false);
223 
224                     $item['content'] = !empty($view)
225                         ? $this->getController()->renderPartial($view, $data, true, $process)
226                         : '';
227                 }
228 
229                 $content = TbHtml::popOption('content', $item, '');
230                 $paneOptions = TbHtml::popOption('paneOptions', $item, array());
231                 $paneOptions['id'] = TbHtml::popOption('id', $item);
232 
233                 $classes = array('tab-pane fade');
234 
235                 if (isset($item['active']) && $item['active'])
236                     $classes[] = 'active in';
237 
238                 $paneOptions = TbHtml::addClassName(implode(' ', $classes), $paneOptions);
239 
240                 $panes[] = CHtml::tag('div', $paneOptions, $content);
241 
242                 $i++; // increment the tab-index
243             }
244 
245             $items[] = $item;
246         }
247 
248         return $items;
249     }
